TeapotApps is an exceptionally lightweight Node.js web framework built from the ground up for modern JavaScript (ESM). It provides clean modularity, lightweight footprint, and intuitive utilities for elegant API design.

$ npm install -g teapotapps

Designed for Simplicity

Enjoy a robust backend structure without unnecessary bloat. Fast to set up, pleasant to program, and highly custom.

Native ESM Support

Uses modern ECMAScript modules out-of-the-box. Clean import/export syntax without bundling, transpile latency, or hacky workarounds.

Modular Codebase

Scaffolds with logical separations for controllers, routing groups, helper libraries, middlewares, and server-side view layers.

Cozy Middlewares

Easily apply CORS, logging, authorization checkouts, or security headers either globally or localized to group routes.

Status 418 Intercept

Built-in customized status middleware that embraces HTTP 418 (I'm a teapot) and filters client-facing response APIs with style.

Developer Tutorial Guide

Learn how to steep your first backend logic inside TeapotApps. Follow the clean syntax below.

quickstart.sh
UTF-8 • TeapotApps Engine
# Install the developer CLI globally $ npm install -g teapotapps # Scaffold your new cozy application $ teapotapps create my-app $ cd my-app # Bootstrap and steeps your server live! $ teapotapps dev # Server starts running at http://localhost:3010 ☕
import { Router } from 'express'; import * as WelcomeController from '../controllers/WelcomeController.js'; import { verifyToken } from '../middlewares/verifyToken.js'; import { routeGroup } from '../core/routeGroup.js'; const router = Router(); // Modular nested routing group with dedicated security middleware! routeGroup(router, "/api", [verifyToken], (api) => { api.get('/wellcome', WelcomeController.apiWellcome); }); export default router;
import { responseApi } from "../libs/RestApiHandler.js"; /** * API welcome controller logic * Returns beautifully formatted client JSON response. */ export const apiWellcome = async (req, res) => { const data = { 'users': 1 }; const meta = { 'saya': 1 }; return responseApi( res, data, meta, "Users fetched successfully!", 0 // Code 418 triggers status 418 ); };
// src/views/Welcome.js export default function Welcome({ name }) { return ` <!DOCTYPE html> <html lang="en"> <head> <title>Steeped Tea</title> <style> body { font-family: sans-serif; background: #090706; color: #f5efe6; text-align: center; } h1 { color: #e6a756; margin-top: 100px; } </style> </head> <body> <h1>☕ Cheers, \${name}!</h1> <p>Minified and served fast in ESM style.</p> </body> </html> `; }
